#include "Game.h"

public Game::Game()
{
	menu = new main_menu();
	lvl = new level();
	lb = new leaderboard();

	active_scene = menu;

}

public Game::~Game()
{
	delete menu;
	delete lvl;
	delete lb;
	
	delete renderer;
	delete input;
}

public void Game::update()
{
	active_scene->update_scene(&input);
	active_scne->render_all(renderer);	
}
